    <script>
        // C# Player 클래스 구조를 그대로 JavaScript로 모사
        const State = {
            Normal: 'Normal (일반)',
            Fatigued: 'Fatigued (피로)',
            GameOver: 'GameOver (게임 오버)',
            Invalid: 'Invalid (잘못됨)'
        };

        class Player {
            constructor() {
                this.MaxHP = 100;
                this._hp = 100;
                
                this.BaseMoveSpeed = 5.0;
                this.BaseDefense = 10.0;
                this.BaseAttackPower = 20.0;

                this.CurrentState = State.Normal;
            }

            // C#의 HP 프로퍼티 역할
            get HP() {
                return this._hp;
            }

            set HP(value) {
                // Math.Clamp(value, 0, MaxHP)
                this._hp = Math.max(0, Math.min(value, this.MaxHP));
                this.UpdatePlayerState();
            }

            // 조건문에 맞춰 상태 업데이트
            UpdatePlayerState() {
                const hpPercentage = (this._hp / this.MaxHP) * 100;

                if (hpPercentage >= 25 && this._hp > 0) {
                    this.CurrentState = State.Normal;
                } else if (hpPercentage < 25 && this._hp > 0) {
                    this.CurrentState = State.Fatigued;
                } else if (this._hp === 0) {
                    this.CurrentState = State.GameOver;
                } else {
                    this.CurrentState = State.Invalid;
                }
            }

            // 상태별 적용 능력치 계산기
            GetCurrentMoveSpeed() {
                if (this.CurrentState === State.GameOver) return 0;
                return this.CurrentState === State.Fatigued ? this.BaseMoveSpeed * 0.5 : this.BaseMoveSpeed;
            }

            GetCurrentDefense() {
                if (this.CurrentState === State.GameOver) return 0;
                return this.CurrentState === State.Fatigued ? this.BaseDefense * 0.7 : this.BaseDefense;
            }

            GetCurrentAttackPower() {
                if (this.CurrentState === State.GameOver) return 0;
                return this.CurrentState === State.Fatigued ? this.BaseAttackPower * 0.6 : this.BaseAttackPower;
            }
        }

        // --- 웹 화면 렌더링 및 제어 코드 ---
        const player = new Player();
        const consoleLog = document.getElementById('console-log');

        function appendLog(text, className = 'log-normal') {
            const div = document.createElement('div');
            div.className = `log-entry ${className}`;
            div.innerText = text;
            consoleLog.appendChild(div);
            consoleLog.scrollTop = consoleLog.scrollHeight;
        }

        function updateUI() {
            const hpPct = (player.HP / player.MaxHP) * 100;
            document.getElementById('hp-val').innerText = `${player.HP} / ${player.MaxHP} (${hpPct}%)`;
            
            const stateEl = document.getElementById('state-val');
            stateEl.innerText = player.CurrentState;
            
            if (player.CurrentState === State.Normal) stateEl.style.color = '#4fc1ff';
            else if (player.CurrentState === State.Fatigued) stateEl.style.color = '#ffeb3b';
            else if (player.CurrentState === State.GameOver) stateEl.style.color = '#ff3333';

            document.getElementById('speed-val').innerText = player.GetCurrentMoveSpeed().toFixed(1);
            document.getElementById('def-val').innerText = player.GetCurrentDefense().toFixed(1);
            document.getElementById('atk-val').innerText = player.GetCurrentAttackPower().toFixed(1);
        }

        // [사례 기능] 데미지 입기
        function takeDamage(amount) {
            appendLog(`\n[사례 실행] 플레이어가 데미지를 ${amount} 입었습니다.`, 'log-system');
            player.HP -= amount;
            
            let logStyle = 'log-normal';
            if (player.CurrentState === State.Fatigued) logStyle = 'log-fatigue';
            else if (player.CurrentState === State.GameOver) logStyle = 'log-gameover';

            appendLog(` -> 현재 HP: ${player.HP}% | 현재 상태: ${player.CurrentState}`, logStyle);
            updateUI();
        }

        // [사례 기능] 플레이어 일반 행동 시도
        function movePlayer() {
            appendLog(`\n[행동 시도] 플레이어가 이동 및 공격을 시도합니다.`, 'log-system');
            
            if (player.CurrentState === State.GameOver) {
                appendLog(`[행동 불가] 게임 오버 상태입니다. 플레이어가 더 이상 행동할 수 없습니다! (게임을 재시작 하십시오.)`, 'log-gameover');
                return;
            }

            appendLog(` -> 플레이어가 성공적으로 움직입니다.`, 'log-normal');
            appendLog(` -> 실시간 적용 능력치: [이동속도: ${player.GetCurrentMoveSpeed()} / 방어력: ${player.GetCurrentDefense()} / 공격력: ${player.GetCurrentAttackPower()}]`, 
                player.CurrentState === State.Fatigued ? 'log-fatigue' : 'log-normal'
            );
        }

        // [사례 기능] 게임 재시작
        function restartGame() {
            appendLog(`\n[시스템] 게임 재시작 신호를 보냅니다.`, 'log-system');
            if (player.CurrentState === State.GameOver) {
                player.HP = 100;
                appendLog(` -> 게임을 성공적으로 재시작하여 일반 상태로 복귀했습니다!`, 'log-normal');
            } else {
                appendLog(` -> [경고] 아직 게임 오버 상태가 아닙니다. 재시작할 수 없습니다.`, 'log-fatigue');
            }
            updateUI();
        }

        function clearConsole() {
            consoleLog.innerHTML = '';
        }

        // 최초 UI 업데이트 및 환영 메시지
        updateUI();
        appendLog("시뮬레이터가 정상 구동되었습니다. 왼쪽 버튼을 눌러 각 사례(상태 변화)를 테스트해보세요.", "log-system");
    </script>
